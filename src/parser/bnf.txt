; GW-BASIC LL(1) Grammar
; Token names match lexer output (see src/lexer/README.md)
;
; Conventions:
;   UPPER_CASE  = terminal (token from lexer)
;   CamelCase   = nonterminal
;   ::=         = production rule
;   |           = alternative (separate lines for LL(1) clarity)
;   e           = epsilon (empty production)
;   NUM         = any NUM_i.../NUM_s.../NUM_d... token
;   VAR         = any VAR_UNK_/VAR_STR_/VAR_INT_/VAR_SNG_/VAR_DBL_ token
;   STR         = any STR_... token
;   REM_TEXT    = any REM_... token

; ============================================================
; Entry point
; ============================================================
;
; The parser always operates on a SINGLE LINE, already tokenized.
; Line numbers (LN__) are handled by the dispatcher before parsing:
;   - LN__ present → store tokenized line in program memory
;   - LN__ absent  → immediate mode: parse + execute
; The parser never sees LN__ tokens.

StmtList      ::= Stmt StmtRest
StmtRest      ::= COLON Stmt StmtRest
StmtRest      ::= e

; ============================================================
; Statements
; ============================================================

Stmt          ::= PrintStmt
Stmt          ::= LetStmt
Stmt          ::= IfStmt
Stmt          ::= ForStmt
Stmt          ::= NextStmt
Stmt          ::= GotoStmt
Stmt          ::= GosubStmt
Stmt          ::= RETURN @RETURN
Stmt          ::= InputStmt
Stmt          ::= ReadStmt
Stmt          ::= DataStmt
Stmt          ::= DimStmt
Stmt          ::= WhileStmt
Stmt          ::= WEND @WEND
Stmt          ::= RemStmt
Stmt          ::= EndStmt
Stmt          ::= STOP @STOP
Stmt          ::= OnStmt
Stmt          ::= DefStmt
Stmt          ::= PokeStmt
Stmt          ::= OpenStmt
Stmt          ::= CloseStmt
Stmt          ::= WriteStmt
Stmt          ::= LineStmt
Stmt          ::= LocateStmt
Stmt          ::= ColorStmt
Stmt          ::= ScreenStmt
Stmt          ::= SoundStmt
Stmt          ::= KeyStmt
Stmt          ::= SwapStmt
Stmt          ::= EraseStmt
Stmt          ::= ErrorStmt
Stmt          ::= ResumeStmt
Stmt          ::= CLS @CLS
Stmt          ::= BEEP @BEEP
Stmt          ::= RANDOMIZE Expr @RANDOMIZE
Stmt          ::= AssignStmt
Stmt          ::= e

; --- PRINT / ? ---
PrintStmt     ::= PRINT PrintList @PEND
PrintList     ::= PrintItem PrintRest
PrintList     ::= e
PrintItem     ::= Expr
PrintItem     ::= TAB( Expr CPAR @PTAB
PrintItem     ::= SPC( Expr CPAR @PSPC
PrintRest     ::= SEMICOLON @PSEMI PrintItem PrintRest
PrintRest     ::= COMA @PTAB PrintItem PrintRest
PrintRest     ::= e

; --- LET / assignment ---
LetStmt       ::= LET AssignStmt
AssignStmt    ::= VarRef EQ Expr @ASSIGN

; --- IF / THEN ---
IfStmt        ::= IF Expr @IF THEN ThenClause ElseClause @ENDIF
ThenClause    ::= NUM @IF_GOTO
ThenClause    ::= StmtList
ElseClause    ::= ELSE @ELSE ThenClause
ElseClause    ::= e

; --- FOR / NEXT ---
ForStmt       ::= FOR VAR EQ Expr TO Expr StepClause @FOR
StepClause    ::= STEP Expr
StepClause    ::= e
NextStmt      ::= NEXT NextVars @NEXT
NextVars      ::= VAR NextVarRest
NextVars      ::= e
NextVarRest   ::= COMA VAR @NEXT NextVarRest
NextVarRest   ::= e

; --- GOTO / GOSUB ---
GotoStmt      ::= GOTO NUM @GOTO
GosubStmt     ::= GOSUB NUM @GOSUB

; --- ON ---
OnStmt        ::= ON Expr OnAction NumList @ON_END
OnAction      ::= GOTO @ON_GOTO
OnAction      ::= GOSUB @ON_GOSUB
NumList       ::= NUM NumListRest
NumListRest   ::= COMA NUM NumListRest
NumListRest   ::= e

; --- INPUT ---
InputStmt     ::= INPUT InputPrompt VarList @INPUT
InputPrompt   ::= STR SEMICOLON @PROMPT
InputPrompt   ::= e
VarList       ::= VarRef VarListRest
VarListRest   ::= COMA VarRef VarListRest
VarListRest   ::= e

; --- READ / DATA ---
ReadStmt      ::= READ VarList @READ
DataStmt      ::= DATA DataList @DATA
DataList      ::= DataItem DataListRest
DataItem      ::= NUM
DataItem      ::= STR
DataItem      ::= MINUS NUM
DataListRest  ::= COMA DataItem DataListRest
DataListRest  ::= e

; --- DIM ---
DimStmt       ::= DIM DimVar DimVarRest
DimVar        ::= VAR OPAR ExprList CPAR @DIM
DimVarRest    ::= COMA DimVar DimVarRest
DimVarRest    ::= e

; --- WHILE ---
WhileStmt     ::= WHILE Expr @WHILE

; --- REM / ' ---
RemStmt       ::= REM REM_TEXT @REM

; --- END ---
EndStmt       ::= END @END

; --- DEF FN ---
DefStmt       ::= DEF DefWhat
DefWhat       ::= FN VAR OPAR VarList CPAR EQ Expr @DEF_FN
DefWhat       ::= DefTypeStmt
DefTypeStmt   ::= DEFINT LetterRange @DEFINT
DefTypeStmt   ::= DEFSNG LetterRange @DEFSNG
DefTypeStmt   ::= DEFDBL LetterRange @DEFDBL
DefTypeStmt   ::= DEFSTR LetterRange @DEFSTR
LetterRange   ::= VAR RangeRest
RangeRest     ::= MINUS VAR
RangeRest     ::= e

; --- POKE ---
PokeStmt      ::= POKE Expr COMA Expr @POKE

; --- OPEN ---
OpenStmt      ::= OPEN Expr OpenMode HASH NUM OpenRecLen @OPEN
OpenMode      ::= FOR OpenAccess
OpenMode      ::= COMA Expr COMA Expr
OpenMode      ::= e
OpenAccess    ::= INPUT
OpenAccess    ::= OUTPUT
OpenAccess    ::= APPEND
OpenRecLen    ::= COMA Expr
OpenRecLen    ::= e

; --- CLOSE ---
CloseStmt     ::= CLOSE CloseList @CLOSE
CloseList     ::= HASH NUM CloseListRest
CloseList     ::= e
CloseListRest ::= COMA HASH NUM CloseListRest
CloseListRest ::= e

; --- WRITE ---
WriteStmt     ::= WRITE WriteTarget ExprList @WRITE
WriteTarget   ::= HASH NUM COMA
WriteTarget   ::= e

; --- LINE ---
LineStmt      ::= LINE LineWhat
LineWhat      ::= INPUT LineInput @LINE_INPUT
LineWhat      ::= OPAR Expr COMA Expr CPAR MINUS OPAR Expr COMA Expr CPAR LineBox @LINE_DRAW
LineInput     ::= InputPrompt VarRef
LineBox       ::= COMA Expr LineBoxFill
LineBox       ::= e
LineBoxFill   ::= COMA VAR
LineBoxFill   ::= e

; --- LOCATE ---
LocateStmt    ::= LOCATE Expr LocateCol @LOCATE
LocateCol     ::= COMA Expr
LocateCol     ::= e

; --- COLOR ---
ColorStmt     ::= COLOR Expr ColorRest @COLOR
ColorRest     ::= COMA Expr ColorBorder
ColorRest     ::= e
ColorBorder   ::= COMA Expr
ColorBorder   ::= e

; --- SCREEN ---
ScreenStmt    ::= SCREEN Expr ScreenRest @SCREEN
ScreenRest    ::= COMA Expr ScreenRest2
ScreenRest    ::= e
ScreenRest2   ::= COMA Expr ScreenRest3
ScreenRest2   ::= e
ScreenRest3   ::= COMA Expr
ScreenRest3   ::= e

; --- SOUND ---
SoundStmt     ::= SOUND Expr COMA Expr @SOUND

; --- KEY ---
KeyStmt       ::= KEY KeyWhat
KeyWhat       ::= ON @KEY_ON
KeyWhat       ::= OFF @KEY_OFF
KeyWhat       ::= LIST @KEY_LIST
KeyWhat       ::= Expr COMA Expr @KEY_SET

; --- SWAP ---
SwapStmt      ::= SWAP VarRef COMA VarRef @SWAP

; --- ERASE ---
EraseStmt     ::= ERASE VAR @ERASE EraseRest
EraseRest     ::= COMA VAR @ERASE EraseRest
EraseRest     ::= e

; --- ERROR / RESUME ---
ErrorStmt     ::= ERROR Expr @ERROR
ResumeStmt    ::= RESUME ResumeTarget
ResumeTarget  ::= NEXT @RESUME_NEXT
ResumeTarget  ::= NUM @RESUME_LINE
ResumeTarget  ::= e @RESUME

; ============================================================
; Expressions (precedence low to high, no left recursion)
; ============================================================
;
; Precedence (GW-BASIC, lowest first):
;   1. IMP
;   2. EQV
;   3. XOR
;   4. OR
;   5. AND
;   6. NOT  (prefix)
;   7. =  <  >  <=  >=  <>   (relational)
;   8. +  -  (additive)
;   9. MOD
;  10. \  (integer division)
;  11. *  /  (multiplicative)
;  12. -  (unary negation)
;  13. ^  (exponentiation, right-assoc)
;  14. primary

Expr          ::= ImpExpr

ImpExpr       ::= EqvExpr ImpRest
ImpRest       ::= IMP EqvExpr @IMP ImpRest
ImpRest       ::= e

EqvExpr       ::= XorExpr EqvRest
EqvRest       ::= EQV XorExpr @EQV EqvRest
EqvRest       ::= e

XorExpr       ::= OrExpr XorRest
XorRest       ::= XOR OrExpr @XOR XorRest
XorRest       ::= e

OrExpr        ::= AndExpr OrRest
OrRest        ::= OR AndExpr @OR OrRest
OrRest        ::= e

AndExpr       ::= NotExpr AndRest
AndRest       ::= AND NotExpr @AND AndRest
AndRest       ::= e

NotExpr       ::= NOT NotExpr @NOT
NotExpr       ::= RelExpr

RelExpr       ::= AddExpr RelRest
RelRest       ::= EQ AddExpr @CMP_EQ RelRest
RelRest       ::= LT AddExpr @CMP_LT RelRest
RelRest       ::= GT AddExpr @CMP_GT RelRest
RelRest       ::= LE AddExpr @CMP_LE RelRest
RelRest       ::= GE AddExpr @CMP_GE RelRest
RelRest       ::= NE AddExpr @CMP_NE RelRest
RelRest       ::= e

AddExpr       ::= ModExpr AddRest
AddRest       ::= PLUS ModExpr @ADD AddRest
AddRest       ::= MINUS ModExpr @SUB AddRest
AddRest       ::= e

ModExpr       ::= IdivExpr ModRest
ModRest       ::= MOD IdivExpr @MOD ModRest
ModRest       ::= e

IdivExpr      ::= MulExpr IdivRest
IdivRest      ::= IDIV MulExpr @IDIV IdivRest
IdivRest      ::= e

MulExpr       ::= UnaryExpr MulRest
MulRest       ::= MUL UnaryExpr @MUL MulRest
MulRest       ::= DIV UnaryExpr @DIV MulRest
MulRest       ::= e

UnaryExpr     ::= MINUS PowExpr @NEG
UnaryExpr     ::= PowExpr

PowExpr       ::= Primary PowRest
PowRest       ::= POW PowExpr @POW
PowRest       ::= e

; ============================================================
; Primary expressions
; ============================================================

Primary       ::= NUM
Primary       ::= STR
Primary       ::= VarOrCall
Primary       ::= OPAR Expr CPAR
Primary       ::= FnCall
Primary       ::= BuiltinCall

; --- Variable reference or array access ---
VarOrCall     ::= VAR ArrayIndex
ArrayIndex    ::= OPAR ExprList CPAR @AIDX
ArrayIndex    ::= e

; --- FN user function call ---
FnCall        ::= FN VAR OPAR ExprList CPAR @CALL_FN

; --- Built-in function calls ---
BuiltinCall   ::= AbsFn
BuiltinCall   ::= IntFn
BuiltinCall   ::= SgnFn
BuiltinCall   ::= SqrFn
BuiltinCall   ::= SinFn
BuiltinCall   ::= CosFn
BuiltinCall   ::= TanFn
BuiltinCall   ::= AtnFn
BuiltinCall   ::= LogFn
BuiltinCall   ::= ExpFn
BuiltinCall   ::= RndFn
BuiltinCall   ::= FixFn
BuiltinCall   ::= CintFn
BuiltinCall   ::= CsngFn
BuiltinCall   ::= CdblFn
BuiltinCall   ::= LenFn
BuiltinCall   ::= ValFn
BuiltinCall   ::= AscFn
BuiltinCall   ::= PeekFn
BuiltinCall   ::= InpFn
BuiltinCall   ::= PosFn
BuiltinCall   ::= FreFn
BuiltinCall   ::= ChrFn
BuiltinCall   ::= StrFn
BuiltinCall   ::= SpaceFn
BuiltinCall   ::= StringFn
BuiltinCall   ::= HexFn
BuiltinCall   ::= OctFn
BuiltinCall   ::= LeftFn
BuiltinCall   ::= RightFn
BuiltinCall   ::= MidFn
BuiltinCall   ::= InstrFn
BuiltinCall   ::= InkeyFn
BuiltinCall   ::= VarPtrFn
BuiltinCall   ::= ERL @FN_ERL
BuiltinCall   ::= ERR @FN_ERR
BuiltinCall   ::= CSRLIN @FN_CSRLIN

; 1-arg numeric functions
AbsFn         ::= ABS OPAR Expr CPAR @FN_ABS
IntFn         ::= INT OPAR Expr CPAR @FN_INT
SgnFn         ::= SGN OPAR Expr CPAR @FN_SGN
SqrFn         ::= SQR OPAR Expr CPAR @FN_SQR
SinFn         ::= SIN OPAR Expr CPAR @FN_SIN
CosFn         ::= COS OPAR Expr CPAR @FN_COS
TanFn         ::= TAN OPAR Expr CPAR @FN_TAN
AtnFn         ::= ATN OPAR Expr CPAR @FN_ATN
LogFn         ::= LOG OPAR Expr CPAR @FN_LOG
ExpFn         ::= EXP OPAR Expr CPAR @FN_EXP
FixFn         ::= FIX OPAR Expr CPAR @FN_FIX
CintFn        ::= CINT OPAR Expr CPAR @FN_CINT
CsngFn        ::= CSNG OPAR Expr CPAR @FN_CSNG
CdblFn        ::= CDBL OPAR Expr CPAR @FN_CDBL
LenFn         ::= LEN OPAR Expr CPAR @FN_LEN
ValFn         ::= VAL OPAR Expr CPAR @FN_VAL
AscFn         ::= ASC OPAR Expr CPAR @FN_ASC
PeekFn        ::= PEEK OPAR Expr CPAR @FN_PEEK
InpFn         ::= INP OPAR Expr CPAR @FN_INP
PosFn         ::= POS OPAR Expr CPAR @FN_POS
FreFn         ::= FRE OPAR Expr CPAR @FN_FRE
ChrFn         ::= CHR$ OPAR Expr CPAR @FN_CHR
StrFn         ::= STR$ OPAR Expr CPAR @FN_STR
SpaceFn       ::= SPACE$ OPAR Expr CPAR @FN_SPACE
HexFn         ::= HEX$ OPAR Expr CPAR @FN_HEX
OctFn         ::= OCT$ OPAR Expr CPAR @FN_OCT
VarPtrFn      ::= VARPTR OPAR VarRef CPAR @FN_VARPTR

; RND can be RND or RND(x)
RndFn         ::= RND RndArgs @FN_RND
RndArgs       ::= OPAR Expr CPAR
RndArgs       ::= e

; 2-arg string functions
LeftFn        ::= LEFT$ OPAR Expr COMA Expr CPAR @FN_LEFT
RightFn       ::= RIGHT$ OPAR Expr COMA Expr CPAR @FN_RIGHT
StringFn      ::= STRING$ OPAR Expr COMA Expr CPAR @FN_STRING

; 2-3 arg
MidFn         ::= MID$ OPAR Expr COMA Expr MidLen CPAR @FN_MID
MidLen        ::= COMA Expr
MidLen        ::= e

; 2-3 arg
InstrFn       ::= INSTR OPAR Expr COMA Expr InstrLen CPAR @FN_INSTR
InstrLen      ::= COMA Expr
InstrLen      ::= e

; No-arg
InkeyFn       ::= INKEY$ @FN_INKEY

; ============================================================
; Shared rules
; ============================================================

VarRef        ::= VAR ArrayIndex

ExprList      ::= Expr ExprListRest
ExprListRest  ::= COMA Expr ExprListRest
ExprListRest  ::= e
